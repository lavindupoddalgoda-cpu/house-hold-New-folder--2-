   673‚Üí   673‚Üí## üé¨ ANIMATION SYSTEM
   674‚Üí   674‚Üí
   675‚Üí   675‚Üí### Library Stack
   676‚Üí   676‚Üí```
   677‚Üí   677‚ÜíFramer Motion ‚Üí Component animations, page transitions, spring physics, whileHover/whileTap
   678‚Üí   678‚ÜíGSAP          ‚Üí Complex timelines, ScrollTrigger reveals, TextPlugin rotating text
   679‚Üí   679‚ÜíLenis         ‚Üí Buttery smooth virtual scroll (replaces native scroll)
   680‚Üí   680‚ÜíTailwind CSS  ‚Üí Shimmer, pulse, float, marquee (via keyframes in config)
   681‚Üí   681‚Üí```
   682‚Üí   682‚Üí
   683‚Üí   683‚Üí### Global Animation Rules
   684‚Üí   684‚Üí1. Every page component: wrap in `<motion.div initial={{ opacity:0 }} animate={{ opacity:1 }} transition={{ duration:0.6 }}>`.
   685‚Üí   685‚Üí2. Scroll reveals: use `whileInView={{ opacity:1, y:0 }}` + `initial={{ opacity:0, y:40 }}` + `viewport={{ once:true }}`.
   686‚Üí   686‚Üí3. Stagger grids: parent `variants` with `staggerChildren:0.08`, children `variants` with `hidden`/`visible`.
   687‚Üí   687‚Üí4. Cards: `whileHover={{ y:-6, transition:{ type:'spring', stiffness:300, damping:30 } }}`.
   688‚Üí   688‚Üí5. Buttons: `whileTap={{ scale:0.96 }}`.
   689‚Üí   689‚Üí6. Exit: wrap unmounting elements in `<AnimatePresence>` with `exit` prop (always needs `key`).
   690‚Üí   690‚Üí7. All GSAP: registered inside `useEffect`, cleaned up in return function.
   691‚Üí   691‚Üí8. Lenis: instantiated in the root `SmoothScrollProvider` client component, synced to GSAP ScrollTrigger.
   692‚Üí   692‚Üí
   693‚Üí   693‚Üí### `src/hooks/useScrollAnimations.ts`
   694‚Üí   694‚Üí```typescript
   695‚Üí   695‚Üí'use client';
   696‚Üí   696‚Üíimport { useEffect } from 'react';
   697‚Üí   697‚Üíimport { gsap }         from 'gsap';
   698‚Üí   698‚Üíimport { ScrollTrigger } from 'gsap/ScrollTrigger';
   699‚Üí   699‚Üíimport Lenis             from '@studio-freight/lenis';
   700‚Üí   700‚Üí
   701‚Üí   701‚Üíexport function useScrollAnimations() {
   702‚Üí   702‚Üí  useEffect(() => {
   703‚Üí   703‚Üí    gsap.registerPlugin(ScrollTrigger);
   704‚Üí   704‚Üí
   705‚Üí   705‚Üí    const lenis = new Lenis({
   706‚Üí   706‚Üí      duration: 1.2,
   707‚Üí   707‚Üí      easing: (t: number) => Math.min(1, 1.001 - Math.pow(2, -10 * t)),
   708‚Üí   708‚Üí    });
   709‚Üí   709‚Üí
   710‚Üí   710‚Üí    lenis.on('scroll', ScrollTrigger.update);
   711‚Üí   711‚Üí    gsap.ticker.add((time) => lenis.raf(time * 1000));
   712‚Üí   712‚Üí    gsap.ticker.lagSmoothing(0);
   713‚Üí   713‚Üí
   714‚Üí   714‚Üí    // Auto-reveal all elements marked data-animate="fade-up"
   715‚Üí   715‚Üí    gsap.utils.toArray('[data-animate="fade-up"]').forEach((el) => {
   716‚Üí   716‚Üí      gsap.fromTo(el, { opacity: 0, y: 50 }, {
   717‚Üí   717‚Üí        opacity: 1, y: 0, duration: 0.9, ease: 'power3.out',
   718‚Üí   718‚Üí        scrollTrigger: { trigger: el, start: 'top 85%', once: true },
   719‚Üí   719‚Üí      });
   720‚Üí   720‚Üí    });
   721‚Üí   721‚Üí
   722‚Üí   722‚Üí    return () => {
   723‚Üí   723‚Üí      lenis.destroy();
   724‚Üí   724‚Üí      ScrollTrigger.getAll().forEach(t => t.kill());
   725‚Üí   725‚Üí      gsap.ticker.remove((time) => lenis.raf(time * 1000));
   726‚Üí   726‚Üí    };
   727‚Üí   727‚Üí  }, []);
   728‚Üí   728‚Üí}
   729‚Üí   729‚Üí```

  1156‚Üí  1156‚Üíimport crypto from 'crypto';
  1157‚Üí  1157‚Üí
  1158‚Üí  1158‚Üíexport async function POST(request: NextRequest) {
  1159‚Üí  1159‚Üí  try {
  1160‚Üí  1160‚Üí    const { orderId, amount, currency } = await request.json();
  1161‚Üí  1161‚Üí    if (!orderId || !amount || !currency)
  1162‚Üí  1162‚Üí      return NextResponse.json({ error: 'Missing required fields' }, { status: 400 });

  1165‚Üí  1165‚Üí    const merchantSecret = process.env.PAYHERE_MERCHANT_SECRET!;

  1187‚Üí  1187‚Üí  try {

  1189‚Üí  1189‚Üí    const fields = Object.fromEntries(data) as Record;

  1219‚Üí  1219‚Üí        });
  1220‚Üí  1220‚Üí      }
  1221‚Üí  1221‚Üí      await batch.commit();
  1222‚Üí  1222‚Üí    }
