## 🧠 ZUSTAND STORE — `src/store/useStore.ts`

```typescript
'use client';
import { create } from 'zustand';
import { persist, createJSONStorage } from 'zustand/middleware';
import type { CartItem, UserProfile } from '@/types';

interface Toast { id: string; message: string; type: 'success'|'error'|'warning'|'info'; }

interface StoreState {
  // ── Cart ──
  cart: CartItem[];
  cartOpen: boolean;
  addToCart: (item: CartItem) => void;
  removeFromCart: (productId: string) => void;
  updateQuantity: (productId: string, qty: number) => void;
  clearCart: () => void;
  setCartOpen: (open: boolean) => void;
  getCartTotal: () => number;
  getCartCount: () => number;
  // ── User ──
  user: UserProfile | null;
  authLoading: boolean;
  wishlist: string[];
  setUser: (user: UserProfile | null) => void;
  setAuthLoading: (loading: boolean) => void;
  addToWishlist: (id: string) => void;
  removeFromWishlist: (id: string) => void;
  setWishlist: (ids: string[]) => void;
  // ── UI ──
  searchOpen: boolean;
  mobileMenuOpen: boolean;
  setSearchOpen: (open: boolean) => void;
  setMobileMenuOpen: (open: boolean) => void;
  toasts: Toast[];
  addToast: (t: Omit) => void;
  removeToast: (id: string) => void;
  // ── Coupon ──
  appliedCoupon: { code: string; type: 'percentage'|'fixed'; value: number } | null;
  applyCoupon: (c: StoreState['appliedCoupon']) => void;
  removeCoupon: () => void;
}

export const useStore = create()(
  persist(
    (set, get) => ({
      // Cart
      cart: [],
      cartOpen: false,
      addToCart: (item) => {
        const existing = get().cart.find(c =>
          c.productId === item.productId &&
          JSON.stringify(c.variant) === JSON.stringify(item.variant)
        );
        if (existing) {
          set({ cart: get().cart.map(c =>
            c.productId === item.productId
              ? { ...c, quantity: Math.min(c.quantity + item.quantity, item.stock) }
              : c
          )});
        } else {
          set({ cart: [...get().cart, item] });
        }
        set({ cartOpen: true });
      },
      removeFromCart: (id) => set({ cart: get().cart.filter(c => c.productId !== id) }),
      updateQuantity: (id, qty) => {
        if (qty < 1) { get().removeFromCart(id); return; }
        set({ cart: get().cart.map(c => c.productId === id ? { ...c, quantity: qty } : c) });
      },
      clearCart: () => set({ cart: [] }),
      setCartOpen: (open) => set({ cartOpen: open }),
      getCartTotal: () => get().cart.reduce((s, i) => s + i.price * i.quantity, 0),
      getCartCount: () => get().cart.reduce((s, i) => s + i.quantity, 0),
      // User
      user: null,
      authLoading: true,
      wishlist: [],
      setUser: (user) => set({ user }),
      setAuthLoading: (loading) => set({ authLoading: loading }),
      addToWishlist: (id) => set({ wishlist: [...new Set([...get().wishlist, id])] }),
      removeFromWishlist: (id) => set({ wishlist: get().wishlist.filter(w => w !== id) }),
```

## 🎬 ANIMATION SYSTEM

### Library Stack
```
Framer Motion → Component animations, page transitions, spring physics, whileHover/whileTap
GSAP          → Complex timelines, ScrollTrigger reveals, TextPlugin rotating text
Lenis         → Buttery smooth virtual scroll (replaces native scroll)
Tailwind CSS  → Shimmer, pulse, float, marquee (via keyframes in config)
```

### Global Animation Rules
1. Every page component: wrap in `<motion.div initial={{ opacity:0 }} animate={{ opacity:1 }} transition={{ duration:0.6 }}>`.
2. Scroll reveals: use `whileInView={{ opacity:1, y:0 }}` + `initial={{ opacity:0, y:40 }}` + `viewport={{ once:true }}`.
3. Stagger grids: parent `variants` with `staggerChildren:0.08`, children `variants` with `hidden`/`visible`.
4. Cards: `whileHover={{ y:-6, transition:{ type:'spring', stiffness:300, damping:30 } }}`.
5. Buttons: `whileTap={{ scale:0.96 }}`.
6. Exit: wrap unmounting elements in `<AnimatePresence>` with `exit` prop (always needs `key`).
7. All GSAP: registered inside `useEffect`, cleaned up in return function.
8. Lenis: instantiated in the root `SmoothScrollProvider` client component, synced to GSAP ScrollTrigger.

### `src/hooks/useScrollAnimations.ts`
```typescript
'use client';
import { useEffect } from 'react';
import { gsap }         from 'gsap';
import { ScrollTrigger } from 'gsap/ScrollTrigger';
import Lenis             from '@studio-freight/lenis';

export function useScrollAnimations() {
  useEffect(() => {
    gsap.registerPlugin(ScrollTrigger);

    const lenis = new Lenis({
      duration: 1.2,
      easing: (t: number) => Math.min(1, 1.001 - Math.pow(2, -10 * t)),
    });

    lenis.on('scroll', ScrollTrigger.update);
    gsap.ticker.add((time) => lenis.raf(time * 1000));
    gsap.ticker.lagSmoothing(0);

    // Auto-reveal all elements marked data-animate="fade-up"
    gsap.utils.toArray('[data-animate="fade-up"]').forEach((el) => {
      gsap.fromTo(el, { opacity: 0, y: 50 }, {
        opacity: 1, y: 0, duration: 0.9, ease: 'power3.out',
        scrollTrigger: { trigger: el, start: 'top 85%', once: true },
      });
    });

    return () => {
      lenis.destroy();
      ScrollTrigger.getAll().forEach(t => t.kill());
      gsap.ticker.remove((time) => lenis.raf(time * 1000));
    };
  }, []);
}
```

## 🔒 PAYHERE INTEGRATION (Secure — Next.js API Routes)

### `src/app/api/payhere/hash/route.ts`
```typescript
import { NextRequest, NextResponse } from 'next/server';
import crypto from 'crypto';

export async function POST(request: NextRequest) {
  try {
    const { orderId, amount, currency } = await request.json();
    if (!orderId || !amount || !currency)
      return NextResponse.json({ error: 'Missing required fields' }, { status: 400 });

    const merchantId     = process.env.PAYHERE_MERCHANT_ID!;
    const merchantSecret = process.env.PAYHERE_MERCHANT_SECRET!;

    const secretHash = crypto.createHash('md5').update(merchantSecret).digest('hex').toUpperCase();
    const hashInput  = `${merchantId}${orderId}${parseFloat(String(amount)).toFixed(2)}${currency}${secretHash}`;
    const hash       = crypto.createHash('md5').update(hashInput).digest('hex').toUpperCase();

    return NextResponse.json({ hash, merchantId });
  } catch (err) {
    console.error('PayHere hash error:', err);
    return NextResponse.json({ error: 'Internal server error' }, { status: 500 });
  }
}
```

### `src/app/api/payhere/notify/route.ts`
```typescript
import { NextRequest, NextResponse } from 'next/server';
import crypto from 'crypto';
import { adminDb } from '@/lib/firebaseAdmin';
import { FieldValue } from 'firebase-admin/firestore';

export async function POST(request: NextRequest) {
  try {
    const data = await request.formData();
    const fields = Object.fromEntries(data) as Record;
    const { merchant_id, order_id, payhere_amount, payhere_currency, status_code, md5sig } = fields;

    // Verify PayHere signature
    const merchantSecret = process.env.PAYHERE_MERCHANT_SECRET!;
    const secretHash = crypto.createHash('md5').update(merchantSecret).digest('hex').toUpperCase();
    const localSig = crypto.createHash('md5')
      .update(`${merchant_id}${order_id}${payhere_amount}${payhere_currency}${status_code}${secretHash}`)
      .digest('hex').toUpperCase();

    if (localSig !== md5sig) {
      console.error('PayHere signature mismatch');
      return NextResponse.json({ error: 'Invalid signature' }, { status: 400 });
    }

    const statusMap: Record = { '2':'paid', '-1':'cancelled', '-2':'failed', '-3':'chargebacked' };
    const newStatus = statusMap[status_code] || 'unknown';

    await adminDb.collection('orders').doc(order_id).update({
      status: newStatus, paymentStatus: newStatus,
      payhereStatus: status_code, updatedAt: new Date(),
    });

    if (status_code === '2') {
      const orderDoc = await adminDb.collection('orders').doc(order_id).get();
      const items = orderDoc.data()?.items || [];
      const batch = adminDb.batch();
      for (const item of items) {
        batch.update(adminDb.collection('products').doc(item.productId), {
          stock: FieldValue.increment(-item.quantity),
        });
      }
      await batch.commit();
    }

    return new NextResponse('OK', { status: 200 });
  } catch (err) {
    console.error('PayHere notify error:', err);
    return NextResponse.json({ error: 'Internal server error' }, { status: 500 });
  }
}
```

## 📋 FIRESTORE RULES — `firestore.rules`

```
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isAuthed()  { return request.auth != null; }
    function getRole()   { return get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role; }
    function isAdmin()   { return isAuthed() && getRole() == 'admin'; }
    function isManager() { return isAuthed() && (getRole() == 'admin' || getRole() == 'manager'); }
    function isOwner(uid){ return isAuthed() && request.auth.uid == uid; }
    function notBanned() { return !get(/databases/$(database)/documents/users/$(request.auth.uid)).data.get('banned', false); }

    match /products/{id} {
      allow read: if resource.data.isActive == true || isAdmin();
      allow create, update, delete: if isAdmin();
    }
    match /orders/{id} {
      allow create: if isAuthed() && notBanned();
      allow read:   if isOwner(resource.data.userId) || isManager();
      allow update: if isManager();
      allow delete: if isAdmin();
    }
    match /users/{uid} {
      allow read:   if isOwner(uid) || isManager();
      allow update: if isOwner(uid) || isAdmin();
      allow create: if isAuthed() && request.auth.uid == uid;
      allow delete: if isAdmin();
      allow list:   if isManager();
      match /addresses/{addrId} { allow read, write: if isOwner(uid); }
      match /wishlist/{prodId}  { allow read, write: if isOwner(uid); }
      match /cart/{any}         { allow read, write: if isOwner(uid); }
    }
    match /categories/{id}  { allow read: if true;       allow write: if isAdmin(); }
    match /banners/{id}     { allow read: if true;       allow write: if isAdmin(); }
    match /promotions/{id}  { allow read: if true;       allow write: if isAdmin(); }
    match /settings/{id}    { allow read: if true;       allow write: if isAdmin(); }
    match /coupons/{code}   { allow read: if isAuthed(); allow write: if isAdmin(); }
    match /reviews/{id} {
      allow read:   if resource.data.isApproved == true || isAdmin();
      allow create: if isAuthed() && notBanned() && request.resource.data.userId == request.auth.uid;
      allow update, delete: if isAdmin();
    }
    match /newsletter/{id}  { allow create: if true; allow read, write: if isAdmin(); }
    match /mail/{id}        { allow read, write: if false; }
  }
}
```
